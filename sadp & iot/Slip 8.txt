interface State {
    void insertQuarter();
    void turnCrank();
    void dispense();
}

class NoQuarterState implements State {
    GumballMachine g;
    NoQuarterState(GumballMachine g){ this.g = g; }

    public void insertQuarter(){
        System.out.println("Quarter inserted");
        g.setState(g.hasQuarter);
    }
    public void turnCrank(){ System.out.println("Insert quarter first"); }
    public void dispense(){ System.out.println("No gumball"); }
}

class HasQuarterState implements State {
    GumballMachine g;
    HasQuarterState(GumballMachine g){ this.g = g; }

    public void insertQuarter(){ System.out.println("Already inserted"); }
    public void turnCrank(){
        System.out.println("Crank turned");
        g.setState(g.sold);
    }
    public void dispense(){ System.out.println("No gumball dispensed"); }
}

class SoldState implements State {
    GumballMachine g;
    SoldState(GumballMachine g){ this.g = g; }

    public void insertQuarter(){ System.out.println("Wait..."); }
    public void turnCrank(){ System.out.println("Already turned"); }
    public void dispense(){
        System.out.println("Gumball released");
        g.count--;
        if(g.count > 0) g.setState(g.noQuarter);
        else g.setState(g.soldOut);
    }
}

class SoldOutState implements State {
    GumballMachine g;
    SoldOutState(GumballMachine g){ this.g = g; }

    public void insertQuarter(){ System.out.println("Sold out"); }
    public void turnCrank(){ System.out.println("Sold out"); }
    public void dispense(){ System.out.println("No gumball"); }
}

public class GumballMachine {
    State noQuarter, hasQuarter, sold, soldOut, state;
    int count;

    public GumballMachine(int count){
        this.count = count;
        noQuarter = new NoQuarterState(this);
        hasQuarter = new HasQuarterState(this);
        sold = new SoldState(this);
        soldOut = new SoldOutState(this);
        state = (count > 0) ? noQuarter : soldOut;
    }

    void setState(State s){ state = s; }
    public void insertQuarter(){ state.insertQuarter(); }
    public void turnCrank(){ state.turnCrank(); state.dispense(); }

    public static void main(String[] args){
        GumballMachine g = new GumballMachine(2);
        g.insertQuarter();
        g.turnCrank();
        g.insertQuarter();
        g.turnCrank();
    }
}

Step 1: Power on and Set Up the Raspberry Pi 
1. Connect the Raspberry Pi to a monitor, keyboard, and mouse. 
2. Insert the microSD card with Raspberry Pi OS (or another Linux OS) installed. 
3. Power on the Raspberry Pi by connecting it to a power source. 
Once booted, log into the Raspberry Pi desktop environment or terminal. 
Step 2: Open the Terminal 
 Open the terminal application from the desktop or press Ctrl + Alt + T. 
Step 3: Execute Common Linux Commands 
You can try executing the following commands, which are commonly used for file and system 
management in Linux. Here’s a brief explanation and example usage of each:
