import java.io.*; 
class LowerCaseInputStream extends FilterInputStream { 
    public LowerCaseInputStream(InputStream in) { 
        super(in); 
    } 
    @Override 
    public int read() throws IOException { 
        int c = super.read(); 
        return (c == -1 ? c : Character.toLowerCase((char) c)); 
    } 
    @Override 
    public int read(byte[] b, int off, int len) throws IOException { 
        int result = super.read(b, off, len); 
        if (result != -1) { 
            for (int i = off; i < off + result; i++) { 
                b[i] = (byte) Character.toLowerCase((char) b[i]); 
            } 
        } 
        return result; 
    } 
}

        } 
    }

class Singleton { 
    private static volatile Singleton instance; 
    private Singleton() { 
        System.out.println("Singleton instance created."); 
    } 
    public static Singleton getInstance() { 
        if (instance == null) {             // First check 
            synchronized (Singleton.class) { 
                if (instance == null) {     // Second check 
                    instance = new Singleton(); 
                } 
            } 
        } 
        return instance; 
    } 
    public void showMessage() { 
        System.out.println("Hello from Singleton!"); 
    } 
}

Q.1 Write a Java Program to implement Adapter pattern for Enumeration iterator 
 
  
import java.util.Enumeration; 
import java.util.Iterator; 
import java.util.Vector; 
 
// Adapter class that adapts Enumeration to Iterator 
class EnumerationIteratorAdapter<T> implements Iterator<T> { 
    private Enumeration<T> enumeration; 
 
    // Constructor 
    public EnumerationIteratorAdapter(Enumeration<T> enumeration) { 
        this.enumeration = enumeration; 
    } 
 
    // hasNext() method from Iterator 
    @Override 
    public boolean hasNext() { 
        return enumeration.hasMoreElements(); 
    } 
 
    // next() method from Iterator 
    @Override 
    public T next() { 
        return enumeration.nextElement(); 
    } 
 
    // remove() method from Iterator is unsupported for Enumeration 
    @Override 
    public void remove() { 
        throw new UnsupportedOperationException("Remove not supported."); 
    } 
} 
 
public class AdapterPatternExample { 
    public static void main(String[] args) { 
        // Creating a Vector and populating it with some data 
        Vector<String> vector = new Vector<>(); 
        vector.add("Element1"); 
        vector.add("Element2"); 
        vector.add("Element3"); 
 
        // Getting an Enumeration from the Vector 
        Enumeration<String> enumeration = vector.elements(); 
 
        // Adapting Enumeration to Iterator 
        Iterator<String> iterator = new EnumerationIteratorAdapter<>(enumeration); 
 
        // Using the adapted Iterator 
        System.out.println("Using Enumeration adapted to Iterator:"); 
        while (iterator.hasNext()) { 
            System.out.println(iterator.next()); 
        } 
    } 
}

    public void medium() {

        System.out.println("Fan speed set to MEDIUM"); 
    } 
    public void low() { 
        speed = LOW; 
        System.out.println("Fan speed set to LOW"); 
    } 
    public void off() { 
        speed = OFF; 
        System.out.println("Fan turned OFF"); 
    } 
    public int getSpeed() { 
        return speed; 
    }

// Command Interface 
interface Command { 
    void execute(); 
    void undo(); 
} 
// Concrete Commands 
class FanHighCommand implements Command { 
    CeilingFan fan; 
    int prevSpeed; 
    public FanHighCommand(CeilingFan fan) { 
        this.fan = fan; 
    } 
    public void execute() { 
        prevSpeed = fan.getSpeed(); 
        fan.high(); 
    } 
    public void undo() { 
        if (prevSpeed == CeilingFan.MEDIUM) fan.medium(); 
        else if (prevSpeed == CeilingFan.LOW) fan.low(); 
        else if (prevSpeed == CeilingFan.OFF) fan.off(); 
    } 
} 
class FanOffCommand implements Command { 
    CeilingFan fan; 
    int prevSpeed; 
    public FanOffCommand(CeilingFan fan) { 
        this.fan = fan; 
    } 
    public void execute() { 
        prevSpeed = fan.getSpeed(); 
        fan.off(); 
    } 
 
    public void undo() { 
        if (prevSpeed == CeilingFan.HIGH) fan.high(); 
        else if (prevSpeed == CeilingFan.MEDIUM) fan.medium(); 
        else if (prevSpeed == CeilingFan.LOW) fan.low(); 
    } 
} 
// Invoker

    Command slot; 
    public void setCommand(Command command) { 
        slot = command; 
    } 
    public void pressButton() { 
        slot.execute(); 
    }

        slot.undo(); 
    }

interface State {
    void insertQuarter();
    void turnCrank();
    void dispense();
}

class NoQuarterState implements State {
    GumballMachine g;
    NoQuarterState(GumballMachine g){ this.g = g; }

    public void insertQuarter(){
        System.out.println("Quarter inserted");
        g.setState(g.hasQuarter);
    }
    public void turnCrank(){ System.out.println("Insert quarter first"); }
    public void dispense(){ System.out.println("No gumball"); }
}

class HasQuarterState implements State {
    GumballMachine g;
    HasQuarterState(GumballMachine g){ this.g = g; }

    public void insertQuarter(){ System.out.println("Already inserted"); }
    public void turnCrank(){
        System.out.println("Crank turned");
        g.setState(g.sold);
    }
    public void dispense(){ System.out.println("No gumball dispensed"); }
}

class SoldState implements State {
    GumballMachine g;
    SoldState(GumballMachine g){ this.g = g; }

    public void insertQuarter(){ System.out.println("Wait..."); }
    public void turnCrank(){ System.out.println("Already turned"); }
    public void dispense(){
        System.out.println("Gumball released");
        g.count--;
        if(g.count > 0) g.setState(g.noQuarter);
        else g.setState(g.soldOut);
    }
}

class SoldOutState implements State {
    GumballMachine g;
    SoldOutState(GumballMachine g){ this.g = g; }

    public void insertQuarter(){ System.out.println("Sold out"); }
    public void turnCrank(){ System.out.println("Sold out"); }
    public void dispense(){ System.out.println("No gumball"); }
}

public class GumballMachine {
    State noQuarter, hasQuarter, sold, soldOut, state;
    int count;

    public GumballMachine(int count){
        this.count = count;
        noQuarter = new NoQuarterState(this);
        hasQuarter = new HasQuarterState(this);
        sold = new SoldState(this);
        soldOut = new SoldOutState(this);
        state = (count > 0) ? noQuarter : soldOut;
    }

    void setState(State s){ state = s; }
    public void insertQuarter(){ state.insertQuarter(); }
    public void turnCrank(){ state.turnCrank(); state.dispense(); }

    public static void main(String[] args){
        GumballMachine g = new GumballMachine(2);
        g.insertQuarter();
        g.turnCrank();
        g.insertQuarter();
        g.turnCrank();
    }
}

import java.util.*;

interface Observer {
    void update(int number);
}

interface Subject {
    void register(Observer o);
    void remove(Observer o);
    void notifyObservers();
}

class NumberSubject implements Subject {
    private List<Observer> observers = new ArrayList<>();
    private int number;

    public void setNumber(int n){
        this.number = n;
        notifyObservers();
    }

    public int getNumber(){ return number; }

    public void register(Observer o){ observers.add(o); }
    public void remove(Observer o){ observers.remove(o); }
    public void notifyObservers(){
        for(Observer o : observers) o.update(number);
    }
}

class HexObserver implements Observer {
    public void update(int number){ System.out.println("Hex: " + Integer.toHexString(number)); }
}

class OctalObserver implements Observer {
    public void update(int number){ System.out.println("Octal: " + Integer.toOctalString(number)); }
}

class BinaryObserver implements Observer {
    public void update(int number){ System.out.println("Binary: " + Integer.toBinaryString(number)); }
}

public class Main {
    public static void main(String[] args){
        NumberSubject subject = new NumberSubject();
        subject.register(new HexObserver());
        subject.register(new OctalObserver());
        subject.register(new BinaryObserver());

        subject.setNumber(25);  // example: prints hex, octal, binary
        subject.setNumber(100);
    }
}

import java.io.FilterInputStream; 
import java.io.IOException; 
import java.io.InputStream; 
class LowerCaseInputStream extends FilterInputStream { 
    protected LowerCaseInputStream(InputStream in) { 
        super(in); 
    } 
    @Override 
    public int read() throws IOException { 
        int c = super.read(); 
        return (c == -1 ? c : Character.toLowerCase((char)c)); 
    } 
    @Override 
    public int read(byte[] b, int off, int len) throws IOException { 
        int result = super.read(b, off, len); 
        for (int i = off; i < off + result; i++) { 
            b[i] = (byte) Character.toLowerCase((char)b[i]); 
        } 
        return result; 
    } 
}
