class CeilingFan { 
    public static final int HIGH = 3; 
    public static final int MEDIUM = 2; 
    public static final int LOW = 1; 
    public static final int OFF = 0; 
    private int speed; 
    public CeilingFan() { 
        speed = OFF; 
    } 
    public void high() { 
        speed = HIGH; 
        System.out.println("Fan speed set to HIGH"); 
    } 
    public void medium() { 
        speed = MEDIUM; 
        System.out.println("Fan speed set to MEDIUM"); 
    } 
    public void low() { 
        speed = LOW; 
        System.out.println("Fan speed set to LOW"); 
    } 
    public void off() { 
        speed = OFF; 
        System.out.println("Fan turned OFF"); 
    } 
    public int getSpeed() { 
        return speed; 
    } 
} 
// Command Interface 
interface Command { 
    void execute(); 
    void undo(); 
} 
// Concrete Commands 
class FanHighCommand implements Command { 
    CeilingFan fan; 
    int prevSpeed; 
    public FanHighCommand(CeilingFan fan) { 
        this.fan = fan; 
    } 
    public void execute() { 
        prevSpeed = fan.getSpeed(); 
        fan.high(); 
    } 
    public void undo() { 
        if (prevSpeed == CeilingFan.MEDIUM) fan.medium(); 
        else if (prevSpeed == CeilingFan.LOW) fan.low(); 
        else if (prevSpeed == CeilingFan.OFF) fan.off(); 
    } 
} 
class FanOffCommand implements Command { 
    CeilingFan fan; 
    int prevSpeed; 
    public FanOffCommand(CeilingFan fan) { 
        this.fan = fan; 
    } 
    public void execute() { 
        prevSpeed = fan.getSpeed(); 
        fan.off(); 
    } 
 
    public void undo() { 
        if (prevSpeed == CeilingFan.HIGH) fan.high(); 
        else if (prevSpeed == CeilingFan.MEDIUM) fan.medium(); 
        else if (prevSpeed == CeilingFan.LOW) fan.low(); 
    } 
} 
// Invoker 
class RemoteControl { 
    Command slot; 
    public void setCommand(Command command) { 
        slot = command; 
    } 
    public void pressButton() { 
        slot.execute(); 
    } 
    public void pressUndo() { 
        slot.undo(); 
    } 
} 
// Test Class 
public class Main { 
    public static void main(String[] args) { 
        RemoteControl remote = new RemoteControl(); 
        CeilingFan fan = new CeilingFan(); 
        Command high = new FanHighCommand(fan); 
        Command off = new FanOffCommand(fan); 
        remote.setCommand(high); 
        remote.pressButton();  // Fan High 
        remote.pressUndo();    // Undo → previous state 
        remote.setCommand(off); 
        remote.pressButton();  // Fan Off 
        remote.pressUndo();    // Undo → previous speed 
    } 
}
